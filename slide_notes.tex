\newcommand{\timeline}[2]{%
  \def\currentphase{#1}%
  \begin{tikzpicture}[
    >=Stealth,
    phase/.style={
      draw=ntnublue!25, fill=ntnusky, rounded corners=2pt,
      minimum height=0.8cm, text=black, font=\small,
      inner xsep=8pt, inner ysep=3pt, text width=6.5cm, align=left,
    },
    current/.style={
      draw=ntnublue, fill=ntnublue!12, rounded corners=2pt,
      minimum height=0.8cm, text=black, font=\small\bfseries,
      inner xsep=8pt, inner ysep=3pt, text width=6.5cm, align=left,
      line width=1.2pt,
    },
    timelabel/.style={font=\scriptsize\color{ntnublue!60}, anchor=east},
  ]
  \def\idx{0}%
  \foreach \dates/\desc in {#2} {%
    \pgfmathtruncatemacro{\idx}{\idx+1}%
    \xdef\idx{\idx}%
    \pgfmathsetmacro{\yval}{-(\idx-1)*1.1}%
    \ifnum\idx=\currentphase
      \node[current, anchor=west] (P\idx) at (0.3,\yval) {\desc};
      \node[timelabel] at (0,\yval) {\dates};
      \node[font=\scriptsize, text=ntnublue] at ($(P\idx.east)+(0.4,0)$) {$\blacktriangleleft$};
    \else
      \node[phase, anchor=west] (P\idx) at (0.3,\yval) {\desc};
      \node[timelabel] at (0,\yval) {\dates};
    \fi
  }%
  \pgfmathtruncatemacro{\lastidx}{\idx}%
  \ifnum\lastidx>1
    \draw[ntnublue!30, line width=1.5pt]
      ([xshift=-0.15cm]P1.west) -- ([xshift=-0.15cm]P\lastidx.west);
  \fi
  \foreach \i in {1,...,\lastidx}{%
    \ifnum\i=\currentphase
      \fill[ntnublue] ([xshift=-0.15cm]P\i.west) circle (3.5pt);
    \else
      \fill[ntnublue!35] ([xshift=-0.15cm]P\i.west) circle (2.5pt);
    \fi
  }%
  \end{tikzpicture}%
}

\subsection{Theory}

\begin{autoframe}{Theoretical Framework}
    \begin{itemize}
        \item Main theoretical result or proposition
        \item Key assumptions and conditions
        \item Intuition behind the result
    \end{itemize}

    \vspace{0.5em}
    The central equation:
    \begin{equation}
        Y_{i,t} = \alpha + \beta X_{i,t} + \gamma Z_{i,t} + \epsilon_{i,t}
    \end{equation}
\end{autoframe}
\note{%
  Walk through the equation term by term:

  \begin{description}
    \item[$Y_{i,t}$] Outcome we want to explain (e.g.\ household spending).
    \item[$X_{i,t}$] Treatment / policy variable (e.g.\ interest rate change).
    \item[$Z_{i,t}$] Controls --- list 2--3 concrete examples.
    \item[$\epsilon_{i,t}$] Everything we cannot observe.
  \end{description}

  \textbf{Identification assumption:}
  $\mathbb{E}[\epsilon_{i,t} \mid X_{i,t}, Z_{i,t}] = 0$.
  Ask: ``When might this fail?'' --- guide students toward omitted
  variable bias and reverse causality.

  \medskip
  \textbf{Timing:} $\sim$5 minutes on this slide.
}

\subsection{Worked Example}

\begin{autoframe}{Worked Example}
    \begin{example}
        Consider the case where $X = 1$ and $Z = 0$.
        Then the model reduces to:
        \[
            Y_{i,t} = \alpha + \beta + \epsilon_{i,t}
        \]
        Interpretation: $\beta$ captures the treatment effect directly.
    \end{example}
\end{autoframe}
\note{%
  Work through the algebra on the board:

  \begin{enumerate}
    \item Start with the full model: $Y = \alpha + \beta X + \gamma Z + \epsilon$.
    \item Plug in $X=1$, $Z=0$ --- show cancellation explicitly.
    \item Result: $Y = \alpha + \beta + \epsilon$.
    \item Interpretation: $\beta$ is the treatment effect for treated units.
  \end{enumerate}

  \textbf{Common mistake:} students forget that $\alpha$ is still present
  and try to interpret $\beta$ as the level of $Y$.

  Ask: ``What would $\beta < 0$ mean in our central bank example?''
}
