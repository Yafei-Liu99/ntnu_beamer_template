\newcounter{yf@gframe}

\AtBeginSubsection{%
  \global\expandafter\xdef
    \csname yf@subname@\the\c@section @\the\c@subsection\endcsname
    {\insertsubsectionhead}%
  \@ifundefined{yf@nsub@\the\c@section}{\global\@namedef{yf@nsub@\the\c@section}{0}}{}%
  \ifnum\@nameuse{yf@nsub@\the\c@section}<\c@subsection\relax
    \global\@namedef{yf@nsub@\the\c@section}{\the\c@subsection}%
  \fi
  \global\@namedef{yf@frames@\the\c@section @\the\c@subsection}{}%
}

\newcommand{\yf@register}{%
  \stepcounter{yf@gframe}%
  \edef\yf@curlist{\@nameuse{yf@frames@\the\c@section @\the\c@subsection}}%
  \ifx\yf@curlist\@empty
    \global\expandafter\xdef
      \csname yf@frames@\the\c@section @\the\c@subsection\endcsname{\theyf@gframe}%
  \else
    \global\expandafter\xdef
      \csname yf@frames@\the\c@section @\the\c@subsection\endcsname
      {\yf@curlist,\theyf@gframe}%
  \fi
}

% Content-frame environment that registers itself
\newenvironment{autoframe}[1]{%
  \yf@register
  \begin{frame}[label=yf@auto@\arabic{yf@gframe}]{#1}%
}{%
  \end{frame}%
}

\AtBeginSection[]{%
  \begin{frame}[label=yf@sec@\the\value{section}]
    \centering\vspace*{\fill}
    {\Huge\bfseries\insertsectionhead}\par\vspace{1em}
    {\large Introduction}
    \vspace*{\fill}
  \end{frame}%
}

\newcommand{\yf@dot}[1]{%
  \ifnum\insertframenumber=\getpagerefnumber{yf@auto@#1}\relax
    \hyperlink{yf@auto@#1}{\textcolor{black}{$\bullet$}}%
  \else
    \hyperlink{yf@auto@#1}{\textcolor{black!45}{$\circ$}}%
  \fi\hspace{2pt}%
}

\def\yf@firstof#1,#2\@nil{\def\yf@firstitem{#1}}

\newcommand{\yf@rsub}[2]{%
  \edef\yf@list{\@nameuse{yf@frames@#1@#2}}%
  \ifx\yf@list\@empty\else
    \expandafter\yf@firstof\yf@list,\@nil
    \hyperlink{yf@auto@\yf@firstitem}{\@nameuse{yf@subname@#1@#2}}~%
    \@for\yf@e:=\yf@list\do{\yf@dot{\yf@e}}%
  \fi
}

\setbeamertemplate{headline}{%
  \leavevmode
  \begin{beamercolorbox}[wd=\paperwidth,ht=3.25ex,dp=1.5ex,leftskip=1em,rightskip=1em]{section in head/foot}%
    \usebeamerfont{section in head/foot}%
    \ifnum\insertsectionnumber>0
      \@ifundefined{yf@nsub@\the\value{section}}{}{%
        \hyperlink{yf@sec@\the\value{section}}{\textbf{\insertsectionhead:}}\hfill
        \@tempcnta=1\relax
        \edef\yf@N{\@nameuse{yf@nsub@\insertsectionnumber}}%
        \@whilenum\@tempcnta<\yf@N\do{%
          \yf@rsub{\insertsectionnumber}{\the\@tempcnta}\hfill
          \advance\@tempcnta by 1\relax
        }%
        \yf@rsub{\insertsectionnumber}{\the\@tempcnta}%
      }%
    \fi
  \end{beamercolorbox}%
}
