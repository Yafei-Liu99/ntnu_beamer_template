\documentclass{beamer}
\usepackage{refcount}

\setbeamertemplate{navigation symbols}{}
\setbeamertemplate{footline}{}
\setbeamercolor{section in head/foot}{bg=structure.fg!20,fg=black}

% --- Dot: filled if current frame, hollow otherwise ---
\newcommand{\navdot}[1]{%
  \ifnum\insertframenumber=\getpagerefnumber{#1}\relax
    \hyperlink{#1}{\textcolor{black}{$\bullet$}}%
  \else
    \hyperlink{#1}{\textcolor{black!45}{$\circ$}}%
  \fi\hspace{2pt}%
}

% --- Auto-intro frame at every \section ---
\AtBeginSection{%
  \begin{frame}[plain,label=yf@sec@\the\value{section}]
    \centering\vspace*{\fill}
    {\Huge\bfseries\insertsectionhead}\par\vspace{1em}
    {\large Introduction}
    \vspace*{\fill}
  \end{frame}%
}

% --- Headline: one \or branch per section ---
\setbeamertemplate{headline}{%
  \leavevmode
  \begin{beamercolorbox}[wd=\paperwidth,ht=3.25ex,dp=1.5ex,leftskip=1em,rightskip=1em]{section in head/foot}%
    \usebeamerfont{section in head/foot}%
    \ifcase\insertsectionnumber\or
      % --- Section A (1) ---
      \hyperlink{yf@sec@1}{\textbf{A:}}\hfill
      \hyperlink{f-Aa1}{a}\,\navdot{f-Aa1}\navdot{f-Aa2}\navdot{f-Aa3}\hfill
      \hyperlink{f-Ab1}{b}\,\navdot{f-Ab1}\navdot{f-Ab2}\hfill
      \hyperlink{f-Ac1}{c}\,\navdot{f-Ac1}%
    \or
      % --- Section B (2) ---
      \hyperlink{yf@sec@2}{\textbf{B:}}\hfill
      \hyperlink{f-Bd1}{d}\,\navdot{f-Bd1}\navdot{f-Bd2}\hfill
      \hyperlink{f-Be1}{e}\,\navdot{f-Be1}\navdot{f-Be2}%
    \or
      % --- Section C (3) ---
      \hyperlink{yf@sec@3}{\textbf{C:}}\hfill
      \hyperlink{f-Cf1}{f}\,\navdot{f-Cf1}%
    \fi
  \end{beamercolorbox}%
}
