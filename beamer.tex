\newcommand{\navdot}[2]{%
  \ifnum\insertframenumber=#2\relax
    \hyperlink{#1}{\textcolor{black}{$\bullet$}}%
  \else
    \hyperlink{#1}{\textcolor{black!45}{$\circ$}}%
  \fi\hspace{2pt}%
}
